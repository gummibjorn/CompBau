class Program{

	void refill(Slurp[] arr){
		int i;
		Slurp tmp;

		while(i<40){
			arr[i] = new Slurp();
			
			if(i%2 != 0){
				arr[i].s = tmp;
			}

			tmp = arr[i];

			i=i+1;
		}
	}

	void Main(){
		Slurp[] arr;
		arr = new Slurp[40];

		int i;
		while(i<10000){
			refill(arr);	
			i=i+1;
		}
	}
}

class Slurp{
	Slurp s;
}
